Quick Start

# Install from source
git clone https://github.com/zjkramer/spaceorbust.git
cd spaceorbust
npm install && npm run build

# Connect your GitHub
node dist/cli/index.js auth <your-token>

# Sync your contributions
node dist/cli/index.js sync

# Check your civilization
node dist/cli/index.js status

# Plan a mission to Mars
node dist/cli/index.js mission plan leo mars_orbit
